# Script for "runscript" program of minicom, to test linux functionality

# Set 15min timeout
# runscript has 2min timeout by default:
# check `timeout` in: https://linux.die.net/man/1/runscript
timeout 900

print "\n[minicom] Waiting to receive test.log..."
# expect has default timeout of 60 seconds
expect {
    "B00000000000000"
    timeout 300
}
! rz

# Wait for the "buildroot login:" message
print "\n[minicom] Waiting for buildroot login..."
expect {
    "buildroot login:"
    timeout 300
}

send "root"

expect {
    "#"
    timeout 10
}

send "rz"
! sz ../../software/tests/dma/dma_demo ../../software/tests/dma/runTest.sh ../../software/tests/dma/drivers/iob_dma.ko ../../software/tests/dma/drivers/iob_axistream_in.ko ../../software/tests/dma/drivers/iob_axistream_out.ko

send "./runTest.sh"

expect {
    "#"
    timeout 100
}

# send "sz test.log"
# ! rz --overwrite

print "\n[minicom] Done Test"

# Exit minicom
! kill `cut -f4 -d' ' /proc/$PPID/stat`
